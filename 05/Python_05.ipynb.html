class Range:
    def __init__(self, start, stop):
        self._current = start
        self._stop = stop
    
    def __iter__(self):
        return self
    
    def __next__(self):
        if self._current < self._stop:
            result = self._current
            self._current += 1
            return result
        else:
            raise StopIteration
            
for e in Range(0, 3):
    print(e)
            
i = Range(0, 3)
print(next(i))
print(next(i))
i = iter(i)

def range(start, stop):
    current = start
    while current < stop:
        reset = yield current
        if reset is not None:
            current = reset - 1
        current += 1

def two_ranges(start, stop):
    for x in range(start, stop):
        yield x
    for x in range(start, stop):
        yield x
